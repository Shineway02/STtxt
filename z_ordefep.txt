declare @t_noa nvarchar(30)
declare @t_pageline int = 7   --------一頁幾行
set @t_noa = case when '#non' = [5] then '' else [5] end

declare @tmp table(
	gno nvarchar(1),
	idno int identity(0,1),
	orderno int,
	pageno int,
	w_totpage int,
	noa nvarchar(30),
	no2 nvarchar(30),
	odate nvarchar(10),
	datea nvarchar(10),
	vtime nvarchar(10),
	custno nvarchar(30),
	comp nvarchar(90),
	tel nvarchar(90),
	fax nvarchar(90),
	addr nvarchar(max),
	a_memo nvarchar(max),
	worker nvarchar(50),
	pno nvarchar(30),
	product nvarchar(90),
	unit nvarchar(15),
	b_memo nvarchar(max),
	mount float,
	weight float,
	price float,
	btotal float,
	money float,
	tax float,
	total float,
	
	creidt float,
	ungqb float,
	unpay float,
	unorde float,
	orde float,
	pay float,
	acredit float
)

insert into @tmp
select
	'0',ROW_NUMBER()over(partition by a.noa order by a.noa),1,0,
	a.noa,b.no2,a.odate,a.datea,a.timea,a.custno,c.comp,a.tel,a.fax,case when isnull(a.addr2,'')='' then a.addr else a.addr2 end,a.memo,a.worker,
	b.productno,b.product,b.unit,b.memo,b.mount,b.weight,b.price,b.total,a.money,a.tax,a.total
	,0,0,0,0,0,0,0
from view_orde a
left join view_ordes b on a.noa = b.noa
left join cust c on a.custno = c.noa
where (a.noa = @t_noa)
order by a.noa,b.no2


declare @a_noa nvarchar(30)
declare @a_count int
declare @idno int
declare @k int = 0 ----差幾頁
declare @pageCount int
declare @orderno int
declare @pageno int

declare cursor_table cursor for
	select noa,count(*),max(orderno) from @tmp group by noa
open cursor_table
fetch next from cursor_table
into @a_noa,@a_count,@orderno
while(@@FETCH_STATUS <> -1)
begin		
	if(@a_count > @t_pageline)
	begin
		set @k = CEILING((cast(@a_count as float)/@t_pageline))
		while(@k > 0)
		begin
			update @tmp set pageno = @k where orderno > ((@k-1)*@t_pageline) and orderno <= (@k*@t_pageline)
			set @k -=1
		end
	end
	fetch next from cursor_table
	into @a_noa,@a_count,@orderno
end
close cursor_table

deallocate cursor_table
update @tmp set orderno = orderno-((pageno-1)*@t_pageline)
declare cursor_table cursor for
	select distinct noa,max(orderno),pageno,min(idno),count(*) from @tmp group by noa,pageno
open cursor_table
fetch next from cursor_table
into @a_noa,@orderno,@pageno,@idno,@a_count
while(@@FETCH_STATUS <> -1)
begin		
	set @k = @t_pageline -(@a_count%@t_pageline)
	set @pageCount = @a_count/@t_pageline
	if(@k < @t_pageline and (@pageCount =0))
	begin
		while(@k > 0)
		begin
			insert into @tmp(gno,orderno,pageno,noa,odate,a_memo,custno)
				select '0',(@orderno+1),@pageno,@a_noa,odate,a_memo,custno from @tmp where idno = @idno
			set @k = @k-1
			set @orderno = @orderno +1
		end
	end
	
	insert into @tmp(gno,orderno,pageno,noa,custno,mount,weight,money,tax,total)
		select '1',(@t_pageline+1),pageno,noa,custno,sum(mount),sum(weight),MAX(money),MAX(tax),MAX(total) from @tmp 
		where gno=0 and noa=@a_noa and pageno=@pageno group by noa,pageno,custno
		
	insert into @tmp(gno,orderno,pageno,noa,custno) 
		select '2',(@t_pageline+2),pageno,noa,custno from @tmp where gno=0 and noa=@a_noa and pageno=@pageno group by noa,pageno,custno
		
	fetch next from cursor_table
	into @a_noa,@orderno,@pageno,@idno,@a_count
end
close cursor_table
deallocate cursor_table

declare cursor_table cursor for
	select distinct noa,count(*) from @tmp group by noa
open cursor_table
fetch next from cursor_table
into @a_noa,@a_count
while(@@FETCH_STATUS <> -1)
begin	
	update @tmp set w_totpage = @a_count/(@t_pageline+2) where noa = @a_noa
	fetch next from cursor_table
	into @a_noa,@a_count
end
close cursor_table
deallocate cursor_table

declare @tmp1 table( 
	noa nvarchar(20), 
	noq nvarchar(10), 
	custno nvarchar(20), 
	stype nvarchar(10),--除了外銷,其實稅額都算5% 
	
	mount1 float, 
	money float, 
	mount2 float,--未出量 
	credit float--使用額度 
) 

insert into @tmp1(noa,noq,custno,stype,mount1,money) 
select a.noa,a.no2,b.custno,b.stype,a.mount,a.total 
from view_ordes a 
left join view_orde b on a.noa=b.noa 
where b.noa is not null 
and b.custno in (select custno from @tmp) 
and a.noa not in (select noa from @tmp)
and isnull(b.enda,0)=0 and isnull(b.cancel,0)=0 
and isnull(a.enda,0)=0 and isnull(a.cancel,0)=0 

update @tmp1 set mount2=isnull(a.mount1,0)-ISNULL(b.mount,0) 
from @tmp1 a 
outer apply(select SUM(isnull(mount,0)) mount from view_vccs where ordeno=a.noa and no2=a.noq) b
 
update @tmp1 set credit = case when stype='3' then (case when mount1=0 then 0 else ROUND(mount2/mount1*money,0) end) 
when mount1=0 or mount2<=0 then 0 else ROUND(mount2/mount1*money*(1.05),0) end 

------vcc 應收帳款 
declare @tmp2 table( 
	custno nvarchar(20), 
	total float, 
	paysale float, 
	credit float--使用額度 
) 

insert into @tmp2(custno,total) 
select custno,SUM(ISNULL(total,0)) total from view_vcc 
where custno in (select custno from @tmp) 
group by custno 

update @tmp2 set total = ISNULL(total,0)+ISNULL(b.tax,0) 
from @tmp2 a 
outer apply (select SUM(ISNULL(tax,0)) tax from vcca where custno=a.custno) b 

insert into @tmp2(custno,total) 
select a.custno,SUM(ISNULL(a.tax,0)) total 
from vcca a 
where custno in (select custno from @tmp) 
and not exists(select * from @tmp2 where custno=a.custno) 
group by a.custno 

update @tmp2 set paysale= b.paysale 
from @tmp2 a 
outer apply(select SUM(ISNULL(c.paysale,0)) paysale from umms c 
left join umm d on d.noa=c.noa where a.custno=d.custno) b 

update @tmp2 set credit = ISNULL(total,0)-ISNULL(paysale,0) 

------UMM 預付 
declare @tmp3 table( 
	custno nvarchar(20), 
	money float, 
	paysale float, 
	unopay float, 
	opay float 
) 
insert into @tmp3(custno,money,paysale) 
select b.custno,SUM(ISNULL(a.money,0)),SUM(ISNULL(a.paysale,0)) 
from umms a left join umm b on a.noa=b.noa 
where b.custno in (select custno from @tmp) group by b.custno 

update @tmp3 set unopay = ISNULL(b.unopay,0) 
from @tmp3 a 
outer apply (select SUM(ISNULL(unopay,0)) unopay from umm where custno=a.custno ) b 

insert into @tmp3(custno,unopay) 
select custno, SUM(ISNULL(unopay,0)) unopay 
from umm where custno in (select custno from @tmp) 
and not exists(select * from @tmp3 where custno=umm.custno) group by custno 

update @tmp3 set opay = ISNULL(money,0)-ISNULL(paysale,0)-ISNULL(unopay,0) 

------GQB 收票未兌現票據 
declare @tmp4 table( 
	custno nvarchar(20), 
	money float 
) 
insert into @tmp4 
select ISNULL(compno,''),SUM(ISNULL(money,0)) 
from gqb where typea='1' 
and (ISNULL(enda,'')='' or upper(enda)='N') 
and (isnull(compno,'') in (select custno from @tmp))
group by ISNULL(compno,'') 

--------------------------------------------------------------------------

update a
set creidt=isnull((select credit  from cust where noa=a.custno),0),
	ungqb=isnull((select money  from @tmp4 where noa=a.custno),0),
	unpay=isnull((select credit  from @tmp2 where noa=a.custno),0),
	unorde=isnull((select sum(credit)  from @tmp1 where noa=a.custno group by custno ),0),
	orde=a.total,
	pay=isnull((select opay  from @tmp3 where noa=a.custno),0)
from @tmp a where gno='1'

update a
set acredit=creidt-ungqb-unpay-unorde-orde+pay
from @tmp a where gno='1'

select
dbo.getComma(mount,[2]) mount
,dbo.getComma(weight,[3]) weight
,dbo.getComma(price,[4]) price
,dbo.getComma(btotal,0) btotal
,dbo.getComma(money,0) money
,dbo.getComma(tax,0) tax
,dbo.getComma(total,0) total
,dbo.getComma(creidt,0) creidt
,dbo.getComma(ungqb,0) ungqb
,dbo.getComma(unpay,0) unpay
,dbo.getComma(unorde,0) unorde
,dbo.getComma(orde,0) orde
,dbo.getComma(pay,0) pay
,dbo.getComma(acredit,0) acredit
,*
,ROW_NUMBER()over(partition by noa order by noa desc,custno,pageno,gno,orderno,odate desc)recno
from @tmp  order by noa desc,recno;
